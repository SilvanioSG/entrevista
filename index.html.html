🎯 Objetivo

Analisar os concursos da Mega-Sena para identificar padrões numéricos raros, com base em modelagem matemática simples. O foco é identificar estruturas que se diferenciam do comportamento aleatório esperado.

🧪 Código Python Utilizado


import pandas as pd
import numpy as np
from datetime import datetime

def diferencas(dezenas):
    return list(np.diff(sorted(dezenas)))

padrao_raro = [6, 6, 12, 12, 20]

url = "https://www.dropbox.com/scl/fi/22qkzvty0xq0g19fxaylf/mega_sena.csv?rlkey=w7c7il0jb7lp9l8cvqykhcz5h&dl=1"
df = pd.read_csv(url)

col_dezenas = [col for col in df.columns if 'dezena' in col.lower()]
df_dezenas = df[col_dezenas].copy()

ocorrencias = []

for i, row in df_dezenas.iterrows():
    dezenas = row.dropna().astype(int).tolist()
    if diferencas(dezenas) == padrao_raro:
        ocorrencias.append(i)

concurso_2888 = df[df['Concurso'] == 2888]
dezenas_2888 = concurso_2888[col_dezenas].values.flatten().astype(int)
dif_2888 = diferencas(dezenas_2888)
